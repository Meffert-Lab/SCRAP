Script:

list_of_species="
hsa
mmu
rno
dre
xtr
cel
dme
ath"

for species in $list_of_species
do

gunzip ${species}_rRNA.fasta.gz

awk '!/^>/ { printf "%s", $0; n = "\n" } /^>/ { print n $0; n = "" } END { printf "%s", n }' ${species}_rRNA.fasta | \
    sed 's/>/>'${species}'_/' | \
    sed 's/ /_/g' | \
    awk '{print $1}' \
    > ${species}_temp_rRNA.fasta

mv ${species}_temp_rRNA.fasta ${species}_rRNA.fasta

done

#Combine individual FASTA files to create rRNA reference FASTA file

touch tmp.combined.fasta

for species in $list_of_species
do

cat tmp.combined.fasta ${species}_rRNA.fasta > tmp2.combined.fasta

mv tmp2.combined.fasta tmp.combined.fasta

done

mv tmp.combined.fasta rRNA.fasta
